  file,
  signal (send) set=(kill,term,int,hup,cont),

  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,

  /run/{,**} rwk,

  /usr/lib/bashio/** ix,

  /var/lib/livekit/** rw,

  /proc/sys/net/** r,

  /dev/null rw,
  /dev/random r,
  /dev/urandom r,
